P(X > 5000) &=& e^{ - \frac{1}{1000} 5000} \\
&=& e^{-5}\\

with $t_1 > t_2$ we can say
\begin{eqnarray*} 
P ( X > t_1 | X > t_2 ) &=& 
\frac{P ( X > t_1 \mbox{ and } X > t_2 ) }{P ( X > t_2 ) }\\
&=& \frac{P ( X > t_1  ) }{P ( X > t_2 ) }\\
&=&  \frac{e^{-\lambda t_1}}{e^{-\lambda t_1}}
\end{eqnarray*}

&=&  \frac{e^{-\frac{1}{1000} \times 5000}}{e^{-\frac{1}{1000} \times 1000}}\\
&=&  \frac{e^{-5}}{e^{-1}}\\
&=&  e^{-4}\\
&=&  0.00182 \\
