\begin{itemize}
    \item P(type I error) = P( X > c ) where X ~ N(100, 0.12 / n)
∴ $(c − 100) / (0.1 / \sqrt{n}) = 1.645$ …… (*)
    \item P(type II error) = P( X < c ) where X ~ N(100.5, 0.52 / n)
∴ $(c − 100.5) / (0.5 / \sqrt{n}) = −1.645$ …… (**)
\end{itemize}
